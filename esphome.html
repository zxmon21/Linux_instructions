ESPHome + Docker Setup

1. Install prerequisites first 
sudo apt update
sudo apt install ca-certificates curl gnupg
2. Add Docker repo + install Docker 
sudo install -m 0755 -d /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
sudo chmod a+r /etc/apt/keyrings/docker.gpg

echo \
"deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
$(. /etc/os-release && echo $VERSION_CODENAME) stable" | \
sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

sudo apt update
sudo apt install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
3. (Optional) Run Docker without sudo 
sudo usermod -aG docker $USER
newgrp docker
4. Allow access to USB devices 
sudo usermod -aG dialout $USER
5. Run ESPHome container with USB access 
docker run -d \
  --name esphome \
  --restart=unless-stopped \
  -v ~/Python/ESPhome_configs:/config \
  -p 6052:6052 \
  --device=/dev/ttyUSB0 \
  esphome/esphome
6. Open ESPHome 
http://localhost:6052
7. Flash via CLI inside running container 
docker exec -it esphome esphome run example.yaml
